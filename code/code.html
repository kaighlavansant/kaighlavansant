                    long long runningTotal = 0;

                    iFile.open("'data.txt"); // opening the files
                
                
                // entering file name and checking that its a file
                
                    do {
                        cout << "Enter file name" << endl;
                        cout << "**";
                        getline (cin, inputFileName);
                        iFile.open(inputFileName);
                        cout << endl;
                // failure of file
                        if (iFile.fail()){
                            iFile.clear();
                            iFile.ignore();
                            cout << "Error: Invalid file name" << endl;// error message
                            continue;
                        }
                    }while (!iFile.is_open());
                   
                    // initializing nums
                    int num1 = 0, num2 = 0;
                
                    // outputting the numbers
                       while (!iFile.eof()) {
                
                        iFile >> num1 >> num2;
                
                            if (iFile.fail()) {
                                iFile.clear();
                                iFile.ignore(100, '\n');
                                cout << "Error in line" << endl;
                                continue;
                            }
                
                            int result = num1 + num2;// initializing results 
                            runningTotal += result;
                
                            cout << num1 << " + " << num2 << " = " << result << endl;
                        }
                
                        cout << "\n";
                        cout << "Running total = " << runningTotal << endl;
                
                    iFile.close(); // REMEMBER to close the file AT THE END
                    return 0;
                    }
